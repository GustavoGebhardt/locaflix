/your-project
│
├── /config               # Arquivos de configuração do aplicativo
│   └── settings.php      # Configurações principais do aplicativo
│
├── /public               # Arquivo público de entrada (index.php)
│   └── index.php         # Ponto de entrada do aplicativo
│
├── /src                  # Código fonte da aplicação
│   ├── /Controllers      # Controladores para a lógica de aplicação
│   ├── /Middleware       # Middleware para processar requests
│   ├── /Models           # Modelos de dados
│   ├── /Routes           # Definições de rotas
│   └── /Services         # Serviços e lógica de negócio
│
├── /templates            # Templates de visualização (se estiver usando algum motor de templates)
│   └── layout.phtml      # Layout principal
│
├── /translations         # Arquivos de tradução (se usar internacionalização)
│
├── /tests                # Testes automatizados
│   └── /unit             # Testes unitários
│
├── /vendor               # Dependências do Composer
│
├── .env                  # Arquivo de variáveis de ambiente
├── composer.json         # Arquivo de configuração do Composer
└── README.md             # Documento de descrição do projeto
